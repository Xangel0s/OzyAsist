Can you show me a basic Go + Gin architecture setup for a new microservice? I need a solid foundation following clean architecture principles.

Certainly. A clean architecture approach in Go with Gin typically involves separating your concerns into distinct layers: Domain, Repository (Data), Usecase (Business Logic), and Delivery (HTTP/Gin Handlers). This makes the service testable, maintainable, and independent of external frameworks.

Here is a simplified standard directory structure and an example of how you might wire up the main entry point.

code main.go 
package main

import (
    "log"
    "os"

    "github.com/gin-gonic/gin"
    "myapp/delivery/http"
    "myapp/repository"
    "myapp/usecase"
    "myapp/infrastructure/database"
)

func main() {
    // 1. Init Database
    dbConn := database.NewPostgresDB(os.Getenv("DB_DSN"))
    if dbConn == nil {
        log.Fatal("Failed to connect to database")
    }

    // 2. Setup Repositories (Data Layer)
    userRepo := repository.NewUserRepository(dbConn)

    // 3. Setup Usecases (Business Logic Layer)
    userUsecase := usecase.NewUserUsecase(userRepo)

    // 4. Setup Delivery (HTTP/Gin Layer)
    router := gin.Default()
    
    // Register routes
    http.NewUserHandler(router, userUsecase)

    // 5. Start Server
    port := os.Getenv("PORT")
    if port == "" {
        port = "8080"
    }
    
    log.Printf("Server starting on port %s", port)
    router.Run(":" + port)
}
